让我们一秒内计算出结果，很明显得写脚本

import json
import re #正则模块
import requests

s = requests.Session()
url1 = 'http://101.200.138.180:10006/get_expression'
url2 = 'http://101.200.138.180:10006/crawler'
r = s.get(url1)
r.encoding = 'utf-8'
r = json.loads(r.text)   #修改编码
print(r)

expression = r['expression']
expression = re.sub(r'×', '*', expression)
result = eval(expression)
print(result)

res = s.get(url2, params={"answer":result})
print(res.text)

flag = []

for i in a1:
    if i in v16:
        flag.append(v16.index(i))
